# BACKEND_PRODUCTION.md — Backend Implementation Guide
# DapurDekaka.com v2
**For:** Cursor AI execution
**Status:** Production Target
**Stack:** Next.js 14 App Router · Neon PostgreSQL · Drizzle ORM · NextAuth v5 · Midtrans · RajaOngkir

---

## CRITICAL RULES — READ FIRST

1. **Never use client prices** — always re-fetch prices from DB in API routes before any calculation
2. **All monetary values are integers** — IDR has no decimals, use `Math.floor()` everywhere
3. **Neon uses HTTP driver** — `@neondatabase/serverless` with `{ ssl: true }`, never `pg` or `postgres` directly
4. **Order numbers are race-condition-safe** — use `order_sequences` table with `ON CONFLICT DO UPDATE`
5. **Webhook handler must be idempotent** — check if `orders.payment_status` is already `paid` before processing
6. **Midtrans item_details must sum exactly to `gross_amount`** — validate before sending or Midtrans rejects
7. **Cold-chain only shipping** — never offer non-frozen couriers (Reguler/Ekonomi), only SiCepat FROZEN, JNE YES, AnterAja FROZEN
8. **Points max 50% of subtotal** — enforce server-side, not client-side
9. **Webhook signature** — verify with SHA512: `order_id + status_code + gross_amount + server_key`
10. **All API routes return standardized JSON** — use `success()`, `unauthorized()`, `serverError()` helpers

### `lib/auth/config.ts`

```typescript
import NextAuth from 'next-auth';
import Google from 'next-auth/providers/google';
import Credentials from 'next-auth/providers/credentials';
import { DrizzleAdapter } from '@auth/drizzle-adapter';
import { db } from '@/lib/db';
import { users, accounts, sessions, verificationTokens } from '@/lib/db/schema';
import bcrypt from 'bcryptjs';
import { eq } from 'drizzle-orm';
import { z } from 'zod';

const credentialsSchema = z.object({
  email: z.string().email(),
  password: z.string().min(6),
});

export const { handlers, signIn, signOut, auth } = NextAuth({
  adapter: DrizzleAdapter(db, {
    usersTable: users,
    accountsTable: accounts,
    sessionsTable: sessions,
    verificationTokensTable: verificationTokens,
  }),
  session: { strategy: 'jwt' },
  providers: [
    Google({
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    }),
    Credentials({
      async authorize(credentials) {
        const parsed = credentialsSchema.safeParse(credentials);
        if (!parsed.success) return null;

        const { email, password } = parsed.data;
        const [user] = await db
          .select()
          .from(users)
          .where(eq(users.email, email))
          .limit(1);

        if (!user || !user.passwordHash) return null;
        if (!user.isActive) return null;

        const isValid = await bcrypt.compare(password, user.passwordHash);
        if (!isValid) return null;

        return {
          id: user.id,
          email: user.email,
          name: user.name,
          image: user.image,
          role: user.role,
        };
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.role = (user as any).role;
        token.userId = user.id;
      }
      return token;
    },
    async session({ session, token }) {
      if (session.user) {
        session.user.role = token.role as string;
        session.user.id = token.userId as string;
      }
      return session;
    },
  },
  pages: {
    signIn: '/masuk',
    error: '/masuk',
  },
});
```

```typescript
import { orders, orderItems, orderStatusHistory, orderSequences, pointsHistory, couponUsages } from '@/lib/db/schema';

/**
 * Generates next order number in format DDK-YYYYMMDD-XXXX
 * Race-condition-safe via atomic UPSERT on order_sequences table
 */
export async function generateOrderNumber(): Promise<string> {
  const dateKey = format(new Date(), 'yyyyMMdd');

  const [row] = await db
    .insert(orderSequences)
    .values({ dateKey, lastSeq: 1 })
    .onConflictDoUpdate({
      target: orderSequences.dateKey,
      set: {
        lastSeq: sql`${orderSequences.lastSeq} + 1`,
        updatedAt: sql`now()`,
      },
    })
    .returning({ lastSeq: orderSequences.lastSeq });

  const seq = String(row.lastSeq).padStart(4, '0');
  return `DDK-${dateKey}-${seq}`;
}
```

### `lib/services/points.service.ts`
```typescript
import { db } from '@/lib/db';
import { users, pointsHistory } from '@/lib/db/schema';
import { eq, sql, lt, and, gt } from 'drizzle-orm';
import { addYears } from 'date-fns';

/**
 * Earn points after order is paid
 * Rate: 1 point per IDR 1,000 (floor division)
 */
export async function earnPoints(
  userId: string,
  orderId: string,
  subtotalIDR: number
): Promise<number> {
  const pointsEarned = Math.floor(subtotalIDR / 1000);
  if (pointsEarned === 0) return 0;

  const expiresAt = addYears(new Date(), 1);

  await db.transaction(async (tx) => {
    await tx.insert(pointsHistory).values({
      userId,
      orderId,
      action: 'earn',
      points: pointsEarned,
      balanceAfter: sql`(SELECT points_balance FROM users WHERE id = ${userId}) + ${pointsEarned}`,
      expiresAt,
      description: `Poin dari pesanan`,
    });

    await tx
      .update(users)
      .set({ pointsBalance: sql`points_balance + ${pointsEarned}` })
      .where(eq(users.id, userId));
  });

  return pointsEarned;
}

/**
 * Validate points redemption request
 * Rules: must have sufficient balance, max 50% of subtotal
 */
export function validatePointsRedemption(
  pointsBalance: number,
  pointsToRedeem: number,
  subtotalIDR: number
): { valid: boolean; error?: string } {
  if (pointsToRedeem <= 0) return { valid: false, error: 'Poin tidak valid' };
  if (pointsToRedeem > pointsBalance) {
    return { valid: false, error: 'Saldo poin tidak cukup' };
  }

  const pointsValueIDR = pointsToRedeem * 10; // 100 points = IDR 1,000 = 10 IDR/point
  const maxAllowedIDR = Math.floor(subtotalIDR * 0.5);

  if (pointsValueIDR > maxAllowedIDR) {
    return {
      valid: false,
      error: `Maksimal penggunaan poin adalah 50% dari subtotal (${Math.floor(maxAllowedIDR / 10)} poin)`,
    };
  }

  return { valid: true };
}

/**
 * Points value: 100 pts = IDR 1,000 (10 IDR per point)
 */
export function pointsToIDR(points: number): number {
  return Math.floor(points / 100) * 1000;
}

export function idrToPoints(idr: number): number {
  return Math.floor(idr / 10);
}

/**
 * Deduct points during checkout — creates history record
 */
export async function redeemPoints(
  userId: string,
  orderId: string,
  points: number,
  tx: any
): Promise<void> {
  const discountIDR = pointsToIDR(points);

  await tx.insert(pointsHistory).values({
    userId,
    orderId,
    action: 'redeem',
    points: -points,
    balanceAfter: sql`(SELECT points_balance FROM users WHERE id = ${userId}) - ${points}`,
    description: `Penukaran poin untuk diskon IDR ${discountIDR.toLocaleString('id-ID')}`,
  });

  await tx
    .update(users)
    .set({ pointsBalance: sql`points_balance - ${points}` })
    .where(eq(users.id, userId));
}

/**
 * Expire points — run as cron job at midnight
 */
export async function expireOverduePoints(): Promise<void> {
  const now = new Date();

  const expiredRows = await db
    .select()
    .from(pointsHistory)
    .where(
      and(
        eq(pointsHistory.action, 'earn'),
        lt(pointsHistory.expiresAt, now),
        gt(pointsHistory.points, 0)
      )
    );

  // Group by user
  const byUser = new Map<string, number>();
  for (const row of expiredRows) {
    const current = byUser.get(row.userId) ?? 0;
    byUser.set(row.userId, current + row.points);
  }

  for (const [userId, totalExpired] of byUser) {
    await db.transaction(async (tx) => {
      const [user] = await tx
        .select({ pointsBalance: users.pointsBalance })
        .from(users)
        .where(eq(users.id, userId));

      const actualExpiry = Math.min(totalExpired, user.pointsBalance);
      if (actualExpiry <= 0) return;

      await tx.insert(pointsHistory).values({
        userId,
        action: 'expire',
        points: -actualExpiry,
        balanceAfter: user.pointsBalance - actualExpiry,
        description: 'Poin kedaluwarsa',
      });

      await tx
        .update(users)
        .set({ pointsBalance: sql`points_balance - ${actualExpiry}` })
        .where(eq(users.id, userId));
    });
  }
}
```

### `lib/services/coupon.service.ts`
```typescript
import { db } from '@/lib/db';
import { coupons, couponUsages } from '@/lib/db/schema';
import { eq, and, sql } from 'drizzle-orm';

interface CouponValidationInput {
  code: string;
  userId?: string;
  subtotalIDR: number;
  shippingCostIDR: number;
  cartItemCount: number;
}

interface CouponValidationResult {
  valid: boolean;
  error?: string;
  discountIDR?: number;
  coupon?: typeof coupons.$inferSelect;
}

export async function validateCoupon(
  input: CouponValidationInput
): Promise<CouponValidationResult> {
  const { code, userId, subtotalIDR, shippingCostIDR } = input;

  const [coupon] = await db
    .select()
    .from(coupons)
    .where(eq(coupons.code, code.toUpperCase().trim()))
    .limit(1);

  if (!coupon) {
    return { valid: false, error: 'Kode kupon tidak ditemukan' };
  }

  if (!coupon.isActive) {
    return { valid: false, error: 'Kupon sudah tidak aktif' };
  }

  const now = new Date();
  if (coupon.startDate && new Date(coupon.startDate) > now) {
    return { valid: false, error: 'Kupon belum berlaku' };
  }
  if (coupon.endDate && new Date(coupon.endDate) < now) {
    return { valid: false, error: 'Kupon sudah kedaluwarsa' };
  }

  if (coupon.maxUses !== null && coupon.usedCount >= coupon.maxUses) {
    return { valid: false, error: 'Kupon sudah mencapai batas penggunaan' };
  }

  if (coupon.minOrderAmount && subtotalIDR < coupon.minOrderAmount) {
    return {
      valid: false,
      error: `Minimal belanja Rp ${coupon.minOrderAmount.toLocaleString('id-ID')} untuk menggunakan kupon ini`,
    };
  }

  // Check per-user limit
  if (userId && coupon.maxUsesPerUser) {
    const [usageCount] = await db
      .select({ count: sql<number>`count(*)` })
      .from(couponUsages)
      .where(and(eq(couponUsages.couponId, coupon.id), eq(couponUsages.userId, userId)));

    if (Number(usageCount.count) >= coupon.maxUsesPerUser) {
      return { valid: false, error: 'Kamu sudah menggunakan kupon ini sebelumnya' };
    }
  }

  // Calculate discount
  let discountIDR = 0;

  switch (coupon.type) {
    case 'percentage':
      discountIDR = Math.floor(subtotalIDR * (coupon.discountValue / 100));
      if (coupon.maxDiscountAmount) {
        discountIDR = Math.min(discountIDR, coupon.maxDiscountAmount);
      }
      break;
    case 'fixed':
      discountIDR = Math.min(coupon.discountValue, subtotalIDR);
      break;
    case 'free_shipping':
      discountIDR = shippingCostIDR;
      break;
    case 'buy_x_get_y':
      // Handled in cart-level logic — discount applied as fixed amount
      discountIDR = coupon.discountValue ?? 0;
      break;
  }

  return { valid: true, discountIDR, coupon };
}
```

```typescript
// COLD-CHAIN ONLY — allowed couriers
export const COLD_CHAIN_COURIERS = [
  { code: 'sicepat', service: 'FROZEN', name: 'SiCepat FROZEN' },
  { code: 'jne', service: 'YES', name: 'JNE YES' },
  { code: 'anteraja', service: 'FROZEN', name: 'AnterAja FROZEN' },
] as const;

const RAJAONGKIR_BASE = 'https://api.rajaongkir.com/starter';

async function rajaOngkirFetch(path: string, options?: RequestInit) {
  const res = await fetch(`${RAJAONGKIR_BASE}${path}`, {
    ...options,
    headers: {
      key: process.env.RAJAONGKIR_API_KEY!,
      'Content-Type': 'application/json',
      ...(options?.headers ?? {}),
    },
  });

  if (!res.ok) {
    throw new Error(`RajaOngkir error: ${res.status}`);
  }

  const data = await res.json();
  return data.rajaongkir;
}

export async function getProvinces() {
  const result = await rajaOngkirFetch('/province');
  return result.results as Array<{ province_id: string; province: string }>;
}

export async function getCitiesByProvince(provinceId: string) {
  const result = await rajaOngkirFetch(`/city?province=${provinceId}`);
  return result.results as Array<{
    city_id: string;
    city_name: string;
    type: string;
    postal_code: string;
  }>;
}

export async function calculateShippingCost(input: {
  originCityId: string;
  destinationCityId: string;
  weightGram: number;
}) {
  const { originCityId, destinationCityId, weightGram } = input;

  // RajaOngkir minimum weight is 1000g
  const weight = Math.max(weightGram, 1000);

  const results: Array<{
    courier: string;
    service: string;
    name: string;
    costIDR: number;
    estimatedDays: string;
  }> = [];

  for (const courier of COLD_CHAIN_COURIERS) {
    try {
      const data = await rajaOngkirFetch('/cost', {
        method: 'POST',
        body: JSON.stringify({
          origin: originCityId,
          destination: destinationCityId,
          weight,
          courier: courier.code,
        }),
      });

      const couriersData = data.results as any[];
      for (const c of couriersData) {
        const service = c.costs.find(
          (s: any) => s.service === courier.service
        );
        if (service) {
          results.push({
            courier: courier.code,
            service: courier.service,
            name: courier.name,
            costIDR: service.cost[0].value,
            estimatedDays: service.cost[0].etd,
          });
        }
      }
    } catch (e) {
      console.warn(`Shipping fetch failed for ${courier.code}:`, e);
    }
  }

  return results;
}
```

```typescript
import Midtrans from 'midtrans-client';

export const snap = new Midtrans.Snap({
  isProduction: process.env.NODE_ENV === 'production',
  serverKey: process.env.MIDTRANS_SERVER_KEY!,
  clientKey: process.env.MIDTRANS_CLIENT_KEY!,
});

export interface MidtransItemDetail {
  id: string;
  name: string;
  price: number;  // integer IDR
  quantity: number;
}

export async function createMidtransTransaction(params: {
  orderId: string;
  orderNumber: string;
  grossAmount: number;
  customerName: string;
  customerEmail: string;
  customerPhone: string;
  itemDetails: MidtransItemDetail[];
}): Promise<{ token: string; redirectUrl: string }> {
  const { orderId, orderNumber, grossAmount, customerName, customerEmail, customerPhone, itemDetails } = params;

  // Validate: item_details must sum to gross_amount
  const itemSum = itemDetails.reduce((sum, item) => sum + item.price * item.quantity, 0);
  if (itemSum !== grossAmount) {
    throw new Error(
      `Midtrans item_details sum (${itemSum}) !== gross_amount (${grossAmount}). Fix calculation.`
    );
  }

  const result = await snap.createTransaction({
    transaction_details: {
      order_id: orderNumber, // Use order number as Midtrans order_id
      gross_amount: grossAmount,
    },
    item_details: itemDetails,
    customer_details: {
      first_name: customerName,
      email: customerEmail,
      phone: customerPhone,
    },
    custom_field1: orderId, // Store internal UUID for webhook lookup
    callbacks: {
      finish: `${process.env.NEXT_PUBLIC_APP_URL}/pesanan/${orderNumber}?payment=finish`,
      unfinish: `${process.env.NEXT_PUBLIC_APP_URL}/pesanan/${orderNumber}?payment=unfinish`,
      error: `${process.env.NEXT_PUBLIC_APP_URL}/pesanan/${orderNumber}?payment=error`,
    },
  });

  return { token: result.token, redirectUrl: result.redirect_url };
}
```

### `app/api/checkout/route.ts`
```typescript

import { auth } from '@/lib/auth/config';
import { db } from '@/lib/db';

  orders, orderItems, orderStatusHistory,
  productVariants, products, users,
  couponUsages, pointsHistory

import { z } from 'zod';
import { success, unauthorized, badRequest, serverError } from '@/lib/helpers/response';
import { generateOrderNumber, addStatusHistory } from '@/lib/services/order.service';
import { validateCoupon } from '@/lib/services/coupon.service';
import { validatePointsRedemption, redeemPoints, pointsToIDR } from '@/lib/services/points.service';
import { calculateShippingCost } from '@/lib/services/shipping.service';
import { createMidtransTransaction } from '@/lib/services/payment.service';
import { ratelimit } from '@/lib/helpers/ratelimit';

const CheckoutSchema = z.object({
  items: z.array(z.object({
    variantId: z.string().uuid(),
    quantity: z.number().int().positive().max(99),
  })).min(1),
  shippingAddress: z.object({
    recipientName: z.string().min(2).max(100),
    phone: z.string().regex(/^(\+62|62|0)8[1-9][0-9]{6,10}$/, 'Format nomor HP tidak valid'),
    province: z.string().min(1),
    provinceId: z.string().min(1),
    city: z.string().min(1),
    cityId: z.string().min(1),
    district: z.string().min(1),
    postalCode: z.string().regex(/^\d{5}$/, 'Kode pos harus 5 digit'),
    fullAddress: z.string().min(10).max(300),
    label: z.enum(['rumah', 'kantor', 'lainnya']).default('rumah'),
  }),
  courierCode: z.enum(['sicepat', 'jne', 'anteraja']),
  courierService: z.enum(['FROZEN', 'YES']),
  couponCode: z.string().optional(),
  pointsToRedeem: z.number().int().min(0).default(0),
  notes: z.string().max(300).optional(),
});

export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session?.user?.id) return unauthorized();

  // Rate limiting: 5 checkout attempts per 10 minutes
  const { success: rateLimitOk } = await ratelimit.checkout.limit(session.user.id);
  if (!rateLimitOk) {
    return badRequest('Terlalu banyak percobaan checkout. Tunggu beberapa menit.');
  }

  let body: z.infer<typeof CheckoutSchema>;
  try {
    body = CheckoutSchema.parse(await req.json());
  } catch (e) {
    return badRequest('Data tidak valid', e);
  }

  try {
    // STEP 1: Fetch live product data from DB (never trust client prices)
    const variantIds = body.items.map((i) => i.variantId);
    const dbVariants = await db
      .select({
        id: productVariants.id,
        productId: productVariants.productId,
        name: productVariants.name,
        sku: productVariants.sku,
        priceIDR: productVariants.priceIDR,
        stock: productVariants.stock,
        weightGram: productVariants.weightGram,
        productName: products.nameId,
        isActive: productVariants.isActive,
      })
      .from(productVariants)
      .leftJoin(products, eq(productVariants.productId, products.id))
      .where(inArray(productVariants.id, variantIds));

    // STEP 2: Validate all items exist and have stock
    for (const item of body.items) {
      const dbVariant = dbVariants.find((v) => v.id === item.variantId);
      if (!dbVariant || !dbVariant.isActive) {
        return badRequest(`Produk tidak tersedia`);
      }
      if (dbVariant.stock < item.quantity) {
        return badRequest(`Stok ${dbVariant.productName} (${dbVariant.name}) tidak cukup. Tersisa ${dbVariant.stock}.`);
      }
    }

    // STEP 3: Calculate subtotal using DB prices
    const subtotalIDR = body.items.reduce((sum, item) => {
      const variant = dbVariants.find((v) => v.id === item.variantId)!;
      return sum + variant.priceIDR * item.quantity;
    }, 0);

    const totalWeightGram = body.items.reduce((sum, item) => {
      const variant = dbVariants.find((v) => v.id === item.variantId)!;
      return sum + variant.weightGram * item.quantity;
    }, 0);

    // STEP 4: Re-validate shipping cost server-side
    const ORIGIN_CITY_ID = process.env.RAJAONGKIR_ORIGIN_CITY_ID!;
    const shippingOptions = await calculateShippingCost({
      originCityId: ORIGIN_CITY_ID,
      destinationCityId: body.shippingAddress.cityId,
      weightGram: totalWeightGram,
    });

    const selectedShipping = shippingOptions.find(
      (s) => s.courier === body.courierCode && s.service === body.courierService
    );

    if (!selectedShipping) {
      return badRequest('Opsi pengiriman tidak valid. Silakan pilih ulang.');
    }

    const shippingCostIDR = selectedShipping.costIDR;

    // STEP 5: Validate coupon (if provided)
    let couponDiscountIDR = 0;
    let couponId: string | null = null;

    if (body.couponCode) {
      const couponResult = await validateCoupon({
        code: body.couponCode,
        userId: session.user.id,
        subtotalIDR,
        shippingCostIDR,
        cartItemCount: body.items.length,
      });

      if (!couponResult.valid) {
        return badRequest(couponResult.error!);
      }

      couponDiscountIDR = couponResult.discountIDR!;
      couponId = couponResult.coupon!.id;
    }

    // STEP 6: Validate points redemption
    let pointsDiscountIDR = 0;

    if (body.pointsToRedeem > 0) {
      const [userData] = await db
        .select({ pointsBalance: users.pointsBalance })
        .from(users)
        .where(eq(users.id, session.user.id));

      const pointsValidation = validatePointsRedemption(
        userData.pointsBalance,
        body.pointsToRedeem,
        subtotalIDR
      );

      if (!pointsValidation.valid) {
        return badRequest(pointsValidation.error!);
      }

      pointsDiscountIDR = pointsToIDR(body.pointsToRedeem);
    }

    // STEP 7: Calculate final total
    const totalDiscountIDR = couponDiscountIDR + pointsDiscountIDR;
    const totalAmountIDR = Math.max(
      subtotalIDR + shippingCostIDR - totalDiscountIDR,
      1000 // Minimum IDR 1,000
    );

    // STEP 8: Generate order number (race-condition-safe)
    const orderNumber = await generateOrderNumber();

    // STEP 9: DB transaction — create order atomically
    const newOrderId = await db.transaction(async (tx) => {
      // Create order
      const [order] = await tx
        .insert(orders)
        .values({
          userId: session.user.id,
          orderNumber,
          status: 'pending_payment',
          paymentStatus: 'pending',
          subtotalIDR,
          shippingCostIDR,
          discountIDR: totalDiscountIDR,
          totalAmountIDR,
          couponId,
          couponDiscountIDR,
          pointsRedeemed: body.pointsToRedeem,
          pointsDiscountIDR,
          shippingCourier: body.courierCode,
          shippingService: body.courierService,
          shippingCourierName: selectedShipping.name,
          estimatedDays: selectedShipping.estimatedDays,
          recipientName: body.shippingAddress.recipientName,
          recipientPhone: body.shippingAddress.phone,
          shippingAddress: body.shippingAddress.fullAddress,
          shippingProvince: body.shippingAddress.province,
          shippingCity: body.shippingAddress.city,
          shippingDistrict: body.shippingAddress.district,
          shippingPostalCode: body.shippingAddress.postalCode,
          notes: body.notes,
        })
        .returning({ id: orders.id });

      // Create order items
      await tx.insert(orderItems).values(
        body.items.map((item) => {
          const variant = dbVariants.find((v) => v.id === item.variantId)!;
          return {
            orderId: order.id,
            productVariantId: item.variantId,
            productId: variant.productId,
            productName: variant.productName!,
            variantName: variant.name,
            sku: variant.sku,
            quantity: item.quantity,
            unitPriceIDR: variant.priceIDR,
            totalPriceIDR: variant.priceIDR * item.quantity,
            weightGram: variant.weightGram,
          };
        })
      );

      // Deduct points if used
      if (body.pointsToRedeem > 0) {
        await redeemPoints(session.user.id, order.id, body.pointsToRedeem, tx);
      }

      // Mark coupon as used (increment used_count)
      if (couponId) {
        await tx
          .update(coupons) // imported from schema
          .set({ usedCount: sql`used_count + 1` })
          .where(eq(coupons.id, couponId));

        await tx.insert(couponUsages).values({
          couponId,
          userId: session.user.id,
          orderId: order.id,
          discountIDR: couponDiscountIDR,
        });
      }

      // Initial status history
      await tx.insert(orderStatusHistory).values({
        orderId: order.id,
        status: 'pending_payment',
        note: 'Pesanan dibuat, menunggu pembayaran',
      });

      return order.id;
    });

    // STEP 10: Create Midtrans transaction (outside DB transaction)
    const itemDetails = body.items.map((item) => {
      const variant = dbVariants.find((v) => v.id === item.variantId)!;
      return {
        id: variant.id,
        name: `${variant.productName} - ${variant.name}`.substring(0, 50),
        price: variant.priceIDR,
        quantity: item.quantity,
      };
    });

    // Add shipping as item_detail
    itemDetails.push({
      id: `SHIPPING-${body.courierCode.toUpperCase()}`,
      name: `Ongkir ${selectedShipping.name}`,
      price: shippingCostIDR,
      quantity: 1,
    });

    // Add discount as negative item_detail (if any)
    if (totalDiscountIDR > 0) {
      itemDetails.push({
        id: 'DISCOUNT',
        name: 'Diskon (Kupon + Poin)',
        price: -totalDiscountIDR,
        quantity: 1,
      });
    }

    const [userInfo] = await db
      .select({ name: users.name, email: users.email, phone: users.phone })
      .from(users)
      .where(eq(users.id, session.user.id));

    const { token: snapToken } = await createMidtransTransaction({
      orderId: newOrderId,
      orderNumber,
      grossAmount: totalAmountIDR,
      customerName: userInfo.name,
      customerEmail: userInfo.email,
      customerPhone: userInfo.phone ?? body.shippingAddress.phone,
      itemDetails,
    });

    // Store snap token on order
    await db
      .update(orders)
      .set({ snapToken })
      .where(eq(orders.id, newOrderId));

    return success({
      orderNumber,
      orderId: newOrderId,
      snapToken,
      totalAmountIDR,
    });
  } catch (e) {
    return serverError('Gagal memproses checkout', e);
  }
}
```

### `app/api/payments/webhook/route.ts`
```typescript
import { NextRequest } from 'next/server';
import { db } from '@/lib/db';
import { orders, orderStatusHistory, inventoryLogs, productVariants } from '@/lib/db/schema';
import { eq, sql } from 'drizzle-orm';
import { createHash } from 'crypto';
import { success, serverError, badRequest } from '@/lib/helpers/response';
import { earnPoints, redeemPoints } from '@/lib/services/points.service';
import { addStatusHistory } from '@/lib/services/order.service';
import { sendOrderConfirmationEmail } from '@/lib/services/notification.service';

function verifyWebhookSignature(
  orderId: string,
  statusCode: string,
  grossAmount: string,
  serverKey: string,
  signatureKey: string
): boolean {
  const hash = createHash('sha512')
    .update(`${orderId}${statusCode}${grossAmount}${serverKey}`)
    .digest('hex');
  return hash === signatureKey;
}

export async function POST(req: NextRequest) {
  let body: any;
  try {
    body = await req.json();
  } catch {
    return badRequest('Invalid JSON');
  }

  const {
    order_id: orderNumber,
    status_code,
    gross_amount,
    signature_key,
    transaction_status,
    payment_type,
    fraud_status,
  } = body;

  // Verify signature
  const isValid = verifyWebhookSignature(
    orderNumber,
    status_code,
    gross_amount,
    process.env.MIDTRANS_SERVER_KEY!,
    signature_key
  );

  if (!isValid) {
    console.error('[WEBHOOK] Invalid signature for order:', orderNumber);
    return badRequest('Invalid signature');
  }

  // Find order by order number
  const [order] = await db
    .select()
    .from(orders)
    .where(eq(orders.orderNumber, orderNumber))
    .limit(1);

  if (!order) {
    console.error('[WEBHOOK] Order not found:', orderNumber);
    return badRequest('Order not found');
  }

  // Determine action
  const isSettled =
    transaction_status === 'settlement' ||
    (transaction_status === 'capture' && fraud_status === 'accept');

  const isFailed =
    transaction_status === 'cancel' ||
    transaction_status === 'deny' ||
    transaction_status === 'expire';

  // IDEMPOTENCY: skip if already processed
  if (isSettled && order.paymentStatus === 'paid') {
    console.log('[WEBHOOK] Already settled, skipping:', orderNumber);
    return success({ message: 'Already processed' });
  }

  if (isSettled) {
    await db.transaction(async (tx) => {
      // Update order status
      await tx
        .update(orders)
        .set({
          paymentStatus: 'paid',
          status: 'paid',
          paymentMethod: payment_type,
          paidAt: new Date(),
        })
        .where(eq(orders.id, order.id));

      // Deduct stock for each item
      const items = await tx
        .select()
        .from(orderItems) // from schema
        .where(eq(orderItems.orderId, order.id));

      for (const item of items) {
        await tx
          .update(productVariants)
          .set({ stock: sql`stock - ${item.quantity}` })
          .where(eq(productVariants.id, item.productVariantId));

        await tx.insert(inventoryLogs).values({
          productVariantId: item.productVariantId,
          action: 'sale',
          quantityChange: -item.quantity,
          note: `Terjual via pesanan ${orderNumber}`,
          referenceId: order.id,
        });
      }

      // Add status history
      await tx.insert(orderStatusHistory).values({
        orderId: order.id,
        status: 'paid',
        note: `Pembayaran berhasil via ${payment_type}`,
      });
    });

    // Earn points (after transaction commits)
    await earnPoints(order.userId!, order.id, order.subtotalIDR);

    // Send confirmation email
    await sendOrderConfirmationEmail(order.id).catch((e) =>
      console.error('[WEBHOOK] Email failed:', e)
    );
  }

  if (isFailed) {
    // Only process if currently pending
    if (order.paymentStatus === 'pending') {
      await db.transaction(async (tx) => {
        await tx
          .update(orders)
          .set({
            paymentStatus: 'failed',
            status: 'cancelled',
          })
          .where(eq(orders.id, order.id));

        // Restore redeemed points if any
        if (order.pointsRedeemed && order.pointsRedeemed > 0) {
          await tx
            .update(users)
            .set({ pointsBalance: sql`points_balance + ${order.pointsRedeemed}` })
            .where(eq(users.id, order.userId!));

          await tx.insert(pointsHistory).values({
            userId: order.userId!,
            orderId: order.id,
            action: 'refund',
            points: order.pointsRedeemed,
            description: 'Pengembalian poin karena pembayaran gagal',
          });
        }

        await tx.insert(orderStatusHistory).values({
          orderId: order.id,
          status: 'cancelled',
          note: `Pembayaran ${transaction_status}`,
        });
      });
    }
  }

  return success({ message: 'Webhook processed' });
}
```

## 11. ADMIN ORDER MANAGEMENT

### `app/api/admin/orders/[id]/route.ts`
```typescript
import { NextRequest } from 'next/server';
import { auth } from '@/lib/auth/config';

const UpdateStatusSchema = z.object({
  status: z.enum(['processing', 'packed', 'shipped', 'delivered', 'cancelled', 'refund_requested', 'refunded']),
  note: z.string().max(500).optional(),
  trackingNumber: z.string().optional(),
  trackingUrl: z.string().url().optional(),
});

export async function PATCH(req: NextRequest, { params }: { params: { id: string } }) {
  const session = await auth();
  if (!session?.user) return unauthorized();
  if (!['superadmin', 'owner', 'warehouse'].includes(session.user.role!)) return forbidden();

  const body = UpdateStatusSchema.safeParse(await req.json());
  if (!body.success) return badRequest('Invalid input', body.error);

  const [order] = await db
    .select()
    .from(orders)
    .where(eq(orders.id, params.id))
    .limit(1);

  if (!order) return notFound('Pesanan tidak ditemukan');

  if (!isValidStatusTransition(order.status, body.data.status)) {
    return badRequest(
      `Status tidak dapat diubah dari "${order.status}" ke "${body.data.status}"`
    );
  }

  await db.transaction(async (tx) => {
    const updateData: any = {
      status: body.data.status,
      updatedAt: new Date(),
    };

    if (body.data.trackingNumber) updateData.trackingNumber = body.data.trackingNumber;
    if (body.data.trackingUrl) updateData.trackingUrl = body.data.trackingUrl;
    if (body.data.status === 'shipped') updateData.shippedAt = new Date();
    if (body.data.status === 'delivered') updateData.deliveredAt = new Date();

    await tx.update(orders).set(updateData).where(eq(orders.id, params.id));

    await tx.insert(orderStatusHistory).values({
      orderId: params.id,
      status: body.data.status as any,
      note: body.data.note ?? null,
      changedBy: session.user.id,
    });

    // If cancelling a PAID order — restore stock
    if (body.data.status === 'cancelled' && order.paymentStatus === 'paid') {
      const items = await tx
        .select()
        .from(orderItems)
        .where(eq(orderItems.orderId, params.id));

      for (const item of items) {
        await tx
          .update(productVariants)
          .set({ stock: sql`stock + ${item.quantity}` })
          .where(eq(productVariants.id, item.productVariantId));
      }
    }
  });

  return success({ message: 'Status pesanan berhasil diperbarui' });
}
```

```typescript
import { Resend } from 'resend';
import { db } from '@/lib/db';
import { orders, orderItems, users } from '@/lib/db/schema';
import { eq } from 'drizzle-orm';
import { OrderConfirmationEmail } from '@/emails/order-confirmation';
import { OrderShippedEmail } from '@/emails/order-shipped';

const resend = new Resend(process.env.RESEND_API_KEY!);

export async function sendOrderConfirmationEmail(orderId: string): Promise<void> {
  const [order] = await db
    .select()
    .from(orders)
    .where(eq(orders.id, orderId));

  if (!order || !order.userId) return;

  const [user] = await db
    .select({ name: users.name, email: users.email })
    .from(users)
    .where(eq(users.id, order.userId));

  const items = await db
    .select()
    .from(orderItems)
    .where(eq(orderItems.orderId, orderId));

  await resend.emails.send({
    from: 'Dapur Dekaka <pesanan@dapurdekaka.com>',
    to: user.email,
    subject: `Pesanan ${order.orderNumber} Dikonfirmasi 🎉`,
    react: OrderConfirmationEmail({ order, user, items }),
  });
}

export async function sendShippingEmail(orderId: string): Promise<void> {
  const [order] = await db.select().from(orders).where(eq(orders.id, orderId));
  if (!order || !order.userId) return;

  const [user] = await db
    .select({ name: users.name, email: users.email })
    .from(users)
    .where(eq(users.id, order.userId));

  await resend.emails.send({
    from: 'Dapur Dekaka <pesanan@dapurdekaka.com>',
    to: user.email,
    subject: `Pesanan ${order.orderNumber} Sedang Dikirim 🚚`,
    react: OrderShippedEmail({ order, user }),
  });
}
```

```typescript
export async function GET(req: NextRequest) {
  const authHeader = req.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
```
